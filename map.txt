<script>
    // map : map is an object that maps keys to values. map cant contain duplicate keys.
    // 2 types : getter and setter

    const cars = new Map([
        ["mercedesEQB", 1800],["AudiQ4", 1600]
    ]);
    cars.set("mercedesEQB", 3000);
    let a = cars.get("AudiQ4");
    document.write(a);
</script>
